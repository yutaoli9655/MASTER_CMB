``` bash
# setup gitbook cli ready
git clone git@github.com:genghuiluo/dw-inv-gitbook.git
cd dw-inv-gitbook
gitbook install

# to setup a web service
gitbook serve &

# to export pdf
gitbook pdf ./ ./dw-cm.pdf
```
